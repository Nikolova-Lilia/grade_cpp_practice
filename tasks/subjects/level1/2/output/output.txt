#include <iostream>
using namespace std;

class Rectangle {
private:
    int width, height;
public:
    Rectangle() : width(0), height(0) {}
    Rectangle(int w, int h) : width(w), height(h) {}
    Rectangle(const Rectangle &r) : width(r.width), height(r.height) {}
    Rectangle& operator=(const Rectangle &r) {
        if (this != &r) {
            width = r.width;
            height = r.height;
        }
        return *this;
    }
    ~Rectangle() {}

    friend ostream& operator<<(ostream &os, const Rectangle &r) {
        os << "Width: " << r.width << ", Height: " << r.height;
        return os;
    }
};

int main() {
    Rectangle r1;
    Rectangle r2(5, 10);
    Rectangle r3 = r2;
    r1 = r3;

    cout << r1 << endl;
    cout << r2 << endl;
    cout << r3 << endl;

    return 0;
}
