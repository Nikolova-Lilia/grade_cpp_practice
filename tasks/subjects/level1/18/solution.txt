#include <iostream>
#include <string>
using namespace std;

class Flower {
private:
    string name;
    int height;
public:
    Flower() : name(""), height(0) {}
    Flower(string n, int h) : name(n), height(h) {}
    Flower(const Flower &f) : name(f.name), height(f.height) {}
    Flower& operator=(const Flower &f) {
        if (this != &f) {
            name = f.name;
            height = f.height;
        }
        return *this;
    }
    ~Flower() {}

    friend ostream& operator<<(ostream &os, const Flower &f) {
        os << "Name: " << f.name << ", Height: " << f.height << " cm";
        return os;
    }
};

int main() {
    Flower f1;
    Flower f2("Rose", 30);
    Flower f3 = f2;
    f1 = f3;

    cout << f1 << endl;
    cout << f2 << endl;
    cout << f3 << endl;
    return 0;
}
