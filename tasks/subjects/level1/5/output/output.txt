#include <iostream>
#include <string>
using namespace std;

class Book {
private:
    string title;
    int pages;
public:
    Book() : title(""), pages(0) {}
    Book(string t, int p) : title(t), pages(p) {}
    Book(const Book &b) : title(b.title), pages(b.pages) {}
    Book& operator=(const Book &b) {
        if (this != &b) {
            title = b.title;
            pages = b.pages;
        }
        return *this;
    }
    ~Book() {}

    friend ostream& operator<<(ostream &os, const Book &b) {
        os << "Title: " << b.title << ", Pages: " << b.pages;
        return os;
    }
};

int main() {
    Book b1;
    Book b2("C++ Basics", 250);
    Book b3 = b2;
    b1 = b3;

    cout << b1 << endl;
    cout << b2 << endl;
    cout << b3 << endl;

    return 0;
}
