#include <iostream>
#include <string>
using namespace std;

class Employee {
private:
    string name;
    double salary;
public:
    Employee() : name(""), salary(0.0) {}
    Employee(string n, double s) : name(n), salary(s) {}
    Employee(const Employee &e) : name(e.name), salary(e.salary) {}
    Employee& operator=(const Employee &e) {
        if (this != &e) {
            name = e.name;
            salary = e.salary;
        }
        return *this;
    }
    ~Employee() {}

    friend ostream& operator<<(ostream &os, const Employee &e) {
        os << "Name: " << e.name << ", Salary: " << e.salary;
        return os;
    }
};

int main() {
    Employee e1;
    Employee e2("Bob", 45000.5);
    Employee e3 = e2;
    e1 = e3;

    cout << e1 << endl;
    cout << e2 << endl;
    cout << e3 << endl;

    return 0;
}
